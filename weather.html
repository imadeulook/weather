// ═══════════════════════════════════════════
// AGENTIC LOOP — получаем погоду Токио и Парижа
// Используем Tool Use + while(true) цикл
// ═══════════════════════════════════════════

// 1. ОПИСАНИЕ ИНСТРУМЕНТОВ для Клода
const tools = [
  {
    name: "get_weather",
    description: "Получить текущую погоду города",
    input_schema: {
      type: "object",
      properties: {
        city: {
          type: "string",
          description: "Название города"
        }
      },
      required: ["city"]
    }
  }
];

// 2. СИМУЛЯЦИЯ РЕАЛЬНОГО API ПОГОДЫ
// (в боевом коде — fetch к openweathermap.org)
async function executeTool(name, input) {
  const weatherDB = {
    "Tokyo": { temp: 23, desc: "ясно", humidity: 61 },
    "Paris": { temp: 15, desc: "облачно", humidity: 73 }
  };
  // Имитируем задержку сети
  await sleep(600);
  return weatherDB[input.city]
    ?? { error: "Город не найден" };
}

// 3. AGENTIC LOOP — сердце агента
async function agentLoop(apiKey, task) {
  const messages = [
    { role: "user", content: task }
  ];
  let iteration = 0;
  const MAX_ITER = 10; // защита от бесконечного цикла

  while (iteration < MAX_ITER) {  // ← ЭТО LOOP
    iteration++;

    // Вызываем Claude API
    const response = await callClaude(apiKey, messages, tools);

    // Добавляем ответ ассистента в историю
    messages.push({
      role: "assistant",
      content: response.content
    });

    // ПРОВЕРЯЕМ stop_reason — главное решение цикла
    if (response.stop_reason === "end_turn") {
      // Задача решена — выходим из цикла
      return { done: true, messages, iteration };
    }

    if (response.stop_reason === "tool_use") {
      // Агент хочет вызвать инструмент
      const toolResults = [];

      for (const block of response.content) {
        if (block.type !== "tool_use") continue;

        // Выполняем инструмент
        const result = await executeTool(
          block.name, block.input
        );

        toolResults.push({
          type: "tool_result",
          tool_use_id: block.id,
          content: JSON.stringify(result)
        });
      }

      // Добавляем результаты в историю
      messages.push({
        role: "user",
        content: toolResults
      });
      // Продолжаем цикл с новыми данными
      continue;
    }
  }
}

// 4. ВЫЗОВ CLAUDE API
async function callClaude(apiKey, messages, tools) {
  const res = await fetch(
    "https://api.anthropic.com/v1/messages",
    {
      method: "POST",
      headers: {
        "Content-Type": "application/json",
        "x-api-key": apiKey,
        "anthropic-version": "2023-06-01"
      },
      body: JSON.stringify({
        model: "claude-sonnet-4-20250514",
        max_tokens: 1024,
        tools,
        messages
      })
    }
  );
  return await res.json();
}

const sleep = ms => new Promise(r => setTimeout(r, ms));
